import { clampV3 } from './scannerTypesV3.js';

function luminance(r, g, b) {
  return r * .2126 + g * .7152 + b * .0722;
}

function percentile(values = [], ratio = .5) {
  if (!values.length) return 0;
  const ordered = [...values].sort((a, b) => a - b);
  return ordered[Math.max(0, Math.min(ordered.length - 1, Math.round((ordered.length - 1) * ratio)))];
}

export function analyzeDocumentColorV1093(image) {
  if (!image?.data || !image.width || !image.height) {
    return {
      mean:0,
      contrast:0,
      colorRatio:0,
      redInkRatio:0,
      glareRatio:0,
      shadowRatio:0,
      preserveColor:false,
    };
  }
  const values = [];
  let total = 0;
  let totalSquared = 0;
  let color = 0;
  let redInk = 0;
  let glare = 0;
  let shadow = 0;
  let count = 0;
  const pixelCount = image.width * image.height;
  const step = Math.max(1, Math.floor(pixelCount / 180000));
  for (let pixel = 0; pixel < pixelCount; pixel += step) {
    const offset = pixel * 4;
    const r = image.data[offset];
    const g = image.data[offset + 1];
    const b = image.data[offset + 2];
    const y = luminance(r, g, b);
    const maximum = Math.max(r, g, b);
    const minimum = Math.min(r, g, b);
    const chroma = maximum - minimum;
    total += y;
    totalSquared += y * y;
    if ((pixel + count) % 7 === 0) values.push(y);
    if (chroma > 28 && y > 45 && y < 245) color += 1;
    if (r > g * 1.18 && r > b * 1.12 && r - Math.max(g, b) > 24 && y > 45) redInk += 1;
    if (y > 248) glare += 1;
    if (y < 62) shadow += 1;
    count += 1;
  }
  const mean = count ? total / count : 0;
  const contrast = count ? Math.sqrt(Math.max(0, totalSquared / count - mean * mean)) : 0;
  const colorRatio = count ? color / count : 0;
  const redInkRatio = count ? redInk / count : 0;
  const glareRatio = count ? glare / count : 0;
  const shadowRatio = count ? shadow / count : 0;
  return {
    mean,
    contrast,
    low:percentile(values, .035),
    high:percentile(values, .975),
    colorRatio,
    redInkRatio,
    glareRatio,
    shadowRatio,
    preserveColor:colorRatio >= .012 || redInkRatio >= .0012,
  };
}

function cleanFromGrayscale(image, metrics) {
  const data = new Uint8ClampedArray(image.data.length);
  const low = Math.max(0, Math.min(190, Number(metrics.low || 24)));
  const high = Math.max(low + 22, Math.min(255, Number(metrics.high || 244)));
  const range = Math.max(22, high - low);
  for (let pixel = 0; pixel < image.width * image.height; pixel += 1) {
    const offset = pixel * 4;
    const source = image.data[offset];
    let normalized = clampV3((source - low) / range) * 255;
    normalized = Math.pow(normalized / 255, .9) * 255;
    if (normalized >= 172) normalized += (255 - normalized) * .48;
    else if (normalized <= 148) normalized *= .84;
    const value = Math.max(0, Math.min(255, Math.round(normalized)));
    data[offset] = value;
    data[offset + 1] = value;
    data[offset + 2] = value;
    data[offset + 3] = 255;
  }
  return { width:image.width, height:image.height, data };
}

function colorBoost(image, metrics) {
  const data = new Uint8ClampedArray(image.data.length);
  const shadowLift = Math.min(22, Math.max(0, metrics.shadowRatio * 70));
  for (let pixel = 0; pixel < image.width * image.height; pixel += 1) {
    const offset = pixel * 4;
    const r = image.data[offset];
    const g = image.data[offset + 1];
    const b = image.data[offset + 2];
    const y = luminance(r, g, b);
    let target = y;
    if (y > 174) target += (255 - y) * .24;
    if (y < 95) target += shadowLift;
    const ratio = target / Math.max(1, y);
    data[offset] = Math.max(0, Math.min(255, Math.round(r * ratio)));
    data[offset + 1] = Math.max(0, Math.min(255, Math.round(g * ratio)));
    data[offset + 2] = Math.max(0, Math.min(255, Math.round(b * ratio)));
    data[offset + 3] = 255;
  }
  return { width:image.width, height:image.height, data };
}

export function autoFixDocumentV1093(source, restored = {}) {
  const metrics = analyzeDocumentColorV1093(source);
  const grayscale = restored.grayscale || source;
  const highContrast = restored.highContrast || grayscale;
  const enhancedColor = colorBoost(restored.color || source, metrics);
  const clean = cleanFromGrayscale(grayscale, {
    ...metrics,
    ...analyzeDocumentColorV1093(grayscale),
  });
  const displayMode = metrics.preserveColor ? 'color-preserve' : 'clean-document';
  const display = metrics.preserveColor ? enhancedColor : clean;
  const ocr = metrics.shadowRatio > .18 || metrics.contrast < 38 ? highContrast : clean;
  return {
    display,
    ocr,
    enhancedColor,
    clean,
    grayscale,
    highContrast,
    metadata:{
      engine:'road-ready-auto-quality-bot-v1093',
      mode:displayMode,
      preserveColor:metrics.preserveColor,
      colorRatio:Number(metrics.colorRatio.toFixed(6)),
      redInkRatio:Number(metrics.redInkRatio.toFixed(6)),
      glareRatio:Number(metrics.glareRatio.toFixed(6)),
      shadowRatio:Number(metrics.shadowRatio.toFixed(6)),
      contrast:Number(metrics.contrast.toFixed(3)),
      illuminationNormalized:true,
      paperWhiteBalanced:true,
      localContrast:true,
      textSharpened:true,
    },
  };
}
