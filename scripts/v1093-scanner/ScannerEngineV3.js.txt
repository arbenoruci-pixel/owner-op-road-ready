import { detectDocumentEdgesV3 } from './EdgeDetectorV3.js';
import { warpPerspectiveV3 } from './PerspectiveEngineV3.js';
import { restoreDocumentV3 } from './RestoreEngineV3.js';
import { selectBestOCRVariantV3 } from './OCRPipelineV3.js';
import {
  decodeImageFileV3,
  imageDataToFileV3,
  rotateImageDataClockwiseV3,
} from './imageUtilsV3.js';
import {
  ROAD_READY_SCANNER_V3_VERSION,
  orderCornersV3,
  rotateCornersClockwiseV3,
  uidV3,
} from './scannerTypesV3.js';
import { boundaryCornersV1093 } from './CurvedBoundaryV1093.js';
import { autoFixDocumentV1093 } from './AutoQualityBotV1093.js';

function captureAsset(kind, file, metadata = {}) {
  return {
    kind,
    file,
    pageIndex:0,
    immutable:kind === 'original',
    primary:metadata.primary === true,
    geometryMode:metadata.geometryMode || '',
    filter:metadata.filter || '',
    createdAt:new Date().toISOString(),
  };
}

function fourCornerInput(input, fallbackCorners) {
  const candidate = Array.isArray(input)
    ? input
    : input && typeof input === 'object'
      ? boundaryCornersV1093(input)
      : fallbackCorners;
  return orderCornersV3(Array.isArray(candidate) && candidate.length >= 4 ? candidate : fallbackCorners);
}

export class ScannerEngineV3 {
  constructor(options = {}) {
    this.maxInputDimension = options.maxInputDimension || 2200;
    this.maxOutputDimension = options.maxOutputDimension || 2400;
  }

  async prepare(file, options = {}) {
    options.onStatus?.('Opening photo…');
    const workingImage = await decodeImageFileV3(file, {
      maxDimension:this.maxInputDimension,
    });
    options.onStatus?.('Finding the four paper corners…');
    const detection = detectDocumentEdgesV3(workingImage);
    const corners = orderCornersV3(detection.corners);
    const reviewFile = await imageDataToFileV3(
      workingImage,
      'road-ready-review-v10931.jpg',
      .94,
    );
    return {
      id:uidV3('scanner_v10931'),
      source:options.source || 'camera',
      originalFile:file,
      reviewFile,
      workingImage,
      corners,
      detection:{
        ...detection,
        corners,
      },
      rotation:0,
      createdAt:new Date().toISOString(),
    };
  }

  async rotate(session, options = {}) {
    options.onStatus?.('Rotating document…');
    const workingImage = rotateImageDataClockwiseV3(session.workingImage);
    const reviewFile = await imageDataToFileV3(
      workingImage,
      'road-ready-review-v10931-rotated.jpg',
      .94,
    );
    const corners = orderCornersV3(rotateCornersClockwiseV3(session.corners));
    return {
      ...session,
      workingImage,
      reviewFile,
      corners,
      detection:{
        ...(session.detection || {}),
        corners,
      },
      rotation:(Number(session.rotation || 0) + 90) % 360,
    };
  }

  async finalize(session, cornersOrLegacyBoundary, options = {}) {
    options.onStatus?.('Correcting four-corner perspective…');
    const corners = fourCornerInput(cornersOrLegacyBoundary, session.corners);
    const geometryMode = 'homography-four-corner-v10931';
    const corrected = warpPerspectiveV3(
      session.workingImage,
      corners,
      {
        maxDimension:this.maxOutputDimension,
        minShortSide:760,
      },
    );

    options.onStatus?.('Balancing shadows and paper color…');
    const restored = restoreDocumentV3(corrected);
    options.onStatus?.('Strengthening text and handwriting…');
    const autoFixed = autoFixDocumentV1093(corrected, restored);
    options.onStatus?.('Choosing the clearest OCR image…');
    const ocr = selectBestOCRVariantV3({
      restoredColor:autoFixed.enhancedColor,
      grayscale:autoFixed.clean,
      highContrast:autoFixed.highContrast,
    });

    const [
      correctedFile,
      displayFile,
      enhancedColorFile,
      cleanFile,
      grayscaleFile,
      highContrastFile,
    ] = await Promise.all([
      imageDataToFileV3(corrected, 'road-ready-perspective-corrected.jpg', .96),
      imageDataToFileV3(autoFixed.display, 'road-ready-auto-fixed-final.jpg', .96),
      imageDataToFileV3(autoFixed.enhancedColor, 'road-ready-enhanced-color.jpg', .95),
      imageDataToFileV3(autoFixed.clean, 'road-ready-auto-clean.jpg', .95),
      imageDataToFileV3(autoFixed.grayscale, 'road-ready-ocr-grayscale.jpg', .94),
      imageDataToFileV3(autoFixed.highContrast, 'road-ready-ocr-high-contrast.jpg', .94),
    ]);

    const byName = {
      restoredColor:enhancedColorFile,
      grayscale:cleanFile,
      highContrast:highContrastFile,
    };
    const ocrFile = byName[ocr.selected?.name]
      || (autoFixed.ocr === autoFixed.highContrast ? highContrastFile : cleanFile);

    const captureAssets = [
      captureAsset('original', session.originalFile, {
        geometryMode:'immutable-source',
      }),
      captureAsset('perspective-corrected', correctedFile, {
        geometryMode,
      }),
      captureAsset('display-final', displayFile, {
        geometryMode,
        filter:'auto-quality-primary-v10931',
        primary:true,
      }),
      captureAsset('enhanced-color', enhancedColorFile, {
        geometryMode,
        filter:'auto-quality-color-v10931',
      }),
      captureAsset('auto-clean', cleanFile, {
        geometryMode,
        filter:'auto-quality-clean-v10931',
      }),
      captureAsset('grayscale-ocr', grayscaleFile, {
        filter:'grayscale-local-contrast-v10931',
      }),
      captureAsset('high-contrast-ocr', highContrastFile, {
        filter:'high-contrast-v10931',
      }),
      captureAsset('ocr-selected', ocrFile, {
        filter:`selected-${ocr.selected?.name || 'clean'}-v10931`,
      }),
    ];

    return {
      originalFile:session.originalFile,
      displayFile,
      ocrFile,
      metadata:{
        source:`road-ready-scanner-v10931-${session.source}`,
        scannerVersion:ROAD_READY_SCANNER_V3_VERSION,
        pageCount:1,
        originalPreserved:true,
        primaryOutput:'displayFile',
        perspectiveCorrected:true,
        captureAssets,
        captureManifest:{
          version:ROAD_READY_SCANNER_V3_VERSION,
          engine:'Road Ready Scanner 0.4.2 four-corner renderer',
          localOnly:true,
          source:session.source,
          rotation:session.rotation,
          boundary:{
            model:'straight-four-corner',
            visibleHandles:4,
            internalBoundaryPoints:4,
            corners,
            geometryMode,
          },
          detection:{
            confidence:session.detection?.confidence || 0,
            method:session.detection?.method || '',
            metrics:session.detection?.metrics || {},
          },
          perspective:{
            method:geometryMode,
            operation:'four-corner-perspective-correction',
            width:corrected.width,
            height:corrected.height,
          },
          restore:{
            ...(restored.metadata || {}),
            autoQuality:autoFixed.metadata,
          },
          persistence:{
            primaryFile:'display-final',
            originalAsset:'original',
            ocrAsset:'ocr-selected',
          },
          ocrSelection:{
            selected:ocr.selected?.name || 'clean',
            scores:ocr.ranked.map(item => ({
              name:item.name,
              score:item.score,
              metrics:item.metrics,
            })),
          },
          completedAt:new Date().toISOString(),
        },
        displayFile,
        ocrFile,
      },
    };
  }
}

export const scannerEngineV3 = new ScannerEngineV3();
